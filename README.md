```ts
import mongoose from "mongoose";

const VideoSchema = new mongoose.Schema({
  title: String,

  hashtags: [String],
  videoUrl: String,
  scheduledAt: Date,
  posted: { type: Boolean, default: false },
});

export default mongoose.models.Video || mongoose.model("Video", VideoSchema);
```
